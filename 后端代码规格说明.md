com.xxx.project
├── config        # 配置类层
│    存放：跨域配置、拦截器配置、Swagger配置、Redis配置、MyBatis配置等
├── controller    # 控制层（接口层）
│    接收前端请求，只负责接收参数、调用service，不写业务逻辑
├── service       # 业务逻辑层
│    处理核心业务代码、事务、逻辑判断
│    ├── service.impl  # 业务实现类
├── entity        # 实体类
│    对应数据库表字段、属性
├── dto           # 数据传输对象
│    前端传给后端、后端返回前端的封装对象
├── vo            # 视图返回对象
│    专门给前端展示用的封装数据
├── mapper        # 持久层 DAO
│    数据库操作接口，配合 MyBatis
├── util          # 工具类
│    通用工具：日期、加密、文件、常量、全局工具
├── exception     # 全局异常处理
│    自定义异常、全局异常捕获
├── interceptor   # 拦截器
│    登录拦截、token校验、请求拦截
├── annotation    # 自定义注解
├── common        # 公共模块
│    统一返回结果、常量类、枚举
└── resources     # 配置文件目录
     application.yml / application.properties
     mybatis mapper xml文件
